  print(num)

my_nums1 <- c(1, 2, 3)

for (num1 in my_nums1) {
  for (num2 in my_nums2) {
    print(num1 + num2)
  }
}

## "if" loop
my_nums <- c(1, 2, 3, 4, 5)

for (num in my_nums) {
  if (num %% 2 == 0) {
    print(num)
  }
}
[1] 2
[1] 4
## "break" loop
my_nums <- c(1, 2, 3, 4, 5)

for (num in my_nums) {
  if (num == 3) {
    break
  }
  print(num)
}
[1] 1
[1] 2
## "continue" loop
my_nums <- c(1, 2, 3, 4, 5)

for (num in my_nums) {
  if (num %% 2 == 1) {
    next
  }
  print(num)
}
[1] 2
[1] 4
  1. Functions: Show that you understand the syntax for declaring your own function, and that you can declare and run your own custom function to accomplish some task (you choose what the function does).
## example 1
my_function <- function(x, y) {
  z <- x + y
  return(z)
}

result <- my_function(3, 5)
print(result) 
[1] 8
## example 2
sum_of_squares <- function(v) {
  sum <- 0

  for (i in v) {
    if (i %% 2 == 0) {
      sum <- sum + i^2
    }
  }
  
  return(sum)
}
